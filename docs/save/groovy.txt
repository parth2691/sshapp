def insertCnts = target.withBatch { stmt ->
      for( i in 1..batchSize ) {
          def param = queue.poll()
          if( param != null ) {
              def sql = "insert into test ( id, name) values ($param.id, $param.name)"
              stmt.addBatch( sql )

          }
      }
  }
