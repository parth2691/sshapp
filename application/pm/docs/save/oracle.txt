新建表空间TSP_PM

     sqlplus system/system@ora92
     @tablespace.txt;

CREATE TABLESPACE TSP_PM DATAFILE
      'D:\ORACLE\ORADATA\ORA92\TSP_PM.DBF' SIZE 100M AUTOEXTEND ON NEXT 20M MAXSIZE UNLIMITED
    LOGGING
    ONLINE
    PERMANENT
    EXTENT MANAGEMENT LOCAL AUTOALLOCATE
    BLOCKSIZE 8K
    SEGMENT SPACE MANAGEMENT AUTO;


    CREATE USER EKUSER
      IDENTIFIED BY EKUSER
      DEFAULT TABLESPACE TSP_SYSTEM
      TEMPORARY TABLESPACE TEMP
      PROFILE DEFAULT
      ACCOUNT UNLOCK;
      -- 3 Roles for EKUSER
      GRANT DBA TO EKUSER;
      GRANT CONNECT TO EKUSER;
      GRANT RESOURCE TO EKUSER;
      ALTER USER EKUSER DEFAULT ROLE ALL;
      -- 1 System Privilege for EKUSER
      GRANT UNLIMITED TABLESPACE TO EKUSER;
      grant select any table to EKUSER;
      grant insert any table to EKUSER;
      grant delete any table to EKUSER;
      grant update any table to EKUSER;
